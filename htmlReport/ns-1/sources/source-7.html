         ActionListener listener = event -> {

             TENIC.inventoryManager.clear();
             TENIC.playerName = "";
             TENIC.playerJob = "";
             TENIC.playerAge = 0;
 
             TENIC.mJFrame.getContentPane().removeAll();
             TENIC.mJFrame.getContentPane().repaint();
             TENIC.showStartScreen();
         };
         new Timer(6000, listener).start();
     }
